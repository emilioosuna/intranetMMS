		SELECT FDI_VENDEDORASIGNADO, FV_DESCRIPCION,
  	(SUM (CASE
	            WHEN FTI_TIPO=12 AND FDI_MONEDA = 2 THEN      (FDI_CANTIDAD * FDI_PRECIOCONDESCUENTO * -1) 
                    ELSE 0 END)+
         SUM (CASE
	            WHEN FTI_TIPO=12 AND FDI_MONEDA = 1 THEN      (FDI_CANTIDAD *FDI_PRECIOCONDESCUENTO / FTI_FACTORREFERENCIA * -1) 
                    ELSE 0 END))+
    	 SUM (CASE
	            WHEN FTI_TIPO=11 AND FDI_MONEDA = 2 THEN      (FDI_CANTIDAD *FDI_PRECIOCONDESCUENTO) 
                    ELSE 0 END) +
         SUM (CASE 
	            WHEN FTI_TIPO=11 AND FDI_MONEDA = 1 THEN      (FDI_CANTIDAD *FDI_PRECIOCONDESCUENTO / FTI_FACTORREFERENCIA) 
                    ELSE 0 END) AS Dolares,
         (SUM (CASE
	            WHEN FTI_TIPO=12 AND FDI_MONEDA = 2 THEN      (FDI_CANTIDAD *FDI_PRECIOCONDESCUENTO * FTI_FACTORREFERENCIA* -1) 
                    ELSE 0 END)+
         SUM (CASE
	            WHEN FTI_TIPO=12 AND FDI_MONEDA = 1 THEN      (FDI_CANTIDAD *FDI_PRECIOCONDESCUENTO * -1) 
                    ELSE 0 END))+
    	 SUM (CASE
	            WHEN FTI_TIPO=11 AND FDI_MONEDA = 2 THEN      (FDI_CANTIDAD *FDI_PRECIOCONDESCUENTO * FTI_FACTORREFERENCIA) 
                    ELSE 0 END) +
         SUM (CASE 
	            WHEN FTI_TIPO=11 AND FDI_MONEDA = 1 THEN      (FDI_CANTIDAD *FDI_PRECIOCONDESCUENTO) 
                    ELSE 0 END) AS Bolivares,
         (SUM (CASE    
	            WHEN FTI_TIPO=12 AND FDI_MONEDA = 2 THEN      (FDI_CANTIDAD * -1) 
                    ELSE 0 END)+
         SUM (CASE     
	            WHEN FTI_TIPO=12 AND FDI_MONEDA = 1 THEN      (FDI_CANTIDAD * -1) 
                    ELSE 0 END))+
         SUM (CASE   
	            WHEN FTI_TIPO=11 AND FDI_MONEDA = 2 THEN      (FDI_CANTIDAD) 
	            ELSE 0 END)+ 
         SUM (CASE   
	            WHEN FTI_TIPO=11 AND FDI_MONEDA = 1 THEN      (FDI_CANTIDAD)  
	            ELSE 0 END) AS Unidades 
FROM SDETALLEVENTA
INNER JOIN SVENDEDORES ON FDI_VENDEDORASIGNADO = FV_CODIGO
INNER JOIN SOPERACIONINV ON  FDI_OPERACION_AUTOINCREMENT=FTI_AUTOINCREMENT 
WHERE FTI_SALDOOPERACION = 0 AND FDI_STATUS=1 AND FDI_FECHAOPERACION  BETWEEN '2024-01-02' AND '2024-01-31'
	GROUP BY FDI_VENDEDORASIGNADO, FV_DESCRIPCION
